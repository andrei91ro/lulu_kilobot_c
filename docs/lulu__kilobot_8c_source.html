67 void process_message() {
68  uint8_t i;
69 
70 
71  uint8_t distance = estimate_distance(&RB_front().dist);
72  uint8_t *data = RB_front().msg.data;
73  uint16_t id = data[INDEX_MSG_OWNER_UID_LOW] | data[INDEX_MSG_OWNER_UID_HIGH] << 8;
74 
75  //search for the robot uid in the current neighbor list
76  for (i = 0; i < MAX_NEIGHBORS; i++)
77  if (mydata->neighbors[i].uid == id)
78  break;
79 
80  //if the sender of this message was not found in the current list of neighbors
81  if (i == MAX_NEIGHBORS)
82  //check for an empty slot
83  for (i = 0; i < MAX_NEIGHBORS; i++)
84  if (mydata->neighbors[i].uid == NO_ID) {
85  //we will remember a new neighbor
86  mydata->nr_neighbors++;
87  break;
88  }
89 
90  //if there was no empty slot
91  if (i == MAX_NEIGHBORS) {
92  printw("kilo_uid: %d - no slot for KB%d", kilo_uid, id);
93  return;
94  }
95 
96  //if we reach this step then i is a valid slot
97  mydata->neighbors[i].uid = id;
98  mydata->neighbors[i].symbolic_id = id - smallest_robot_uid;
99  //if there is no previous recording of the distance to this neighbor
100  if (mydata->neighbors[i].distance == mydata->neighbors[i].distance_prev &&
101  mydata->neighbors[i].distance == 0)
102  //save current distance in both fields
103  mydata->neighbors[i].distance = mydata->neighbors[i].distance_prev = distance;
104  else {
105  //save previous distance
106  mydata->neighbors[i].distance_prev = mydata->neighbors[i].distance;
107  //save current distance
108  mydata->neighbors[i].distance = distance;
109  }
110 
111  //set the moment in the future when the robot will forget about this neighbor
112  mydata->neighbors[i].timexp_forget = kilo_ticks + FORGET_NEIGHBOR_INTERVAL;
113 }

219 void loop() {
220 #ifdef PCOL_SIM
221  printi("\nLOOP for robot %d\n-------------------------\n", kilo_uid);
222 #endif
223  //if the previous step was the last one
224  if (mydata->sim_result == SIM_STEP_RESULT_NO_MORE_EXECUTABLES) {
225  //mark the end of the simulation and exit
226  set_color(RGB(0, 2, 0));
227  set_motion(MOTION_STOP);
228  printi("kilo_uid %d: NO_MORE_EXEC", kilo_uid);
229  return;
230  } else
231  // if the previous step resulted in an error
232  if (mydata->sim_result == SIM_STEP_RESULT_ERROR) {
233  set_color(RGB(2, 0, 0));
234  set_motion(MOTION_STOP);
235  //we don't print any error message here because we wouldn't be able to read the LULU error because this continuous print
236  return;
237  }
238 
239  //remove old neighbors that haven't sent any recent message
240  forget_neighbors();
241 
242  //process the entire received message buffer
243  while (!RB_empty()) {
244  process_message();
245  RB_popfront();
246  }
247 
248 #ifdef USING_ID_SECURITY
249  if (kilo_uid == STRANGER_UID) {
250  set_color(RGB(3, 0, 3)); //magenta
251  return;
252  }
253 #endif
254  //transform sensor input into symbolic objects
255  procInputModule();

257  //transform symbolic objects into effector commands
